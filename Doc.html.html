<script>
  let turnoSeleccionado = '';

  // Mostrar fecha actual
  const now = new Date();
  const opciones = { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' };
  const fechaLarga = now.toLocaleDateString('es-AR', opciones);
  const fechaCorta = now.toLocaleDateString('es-AR');

  document.getElementById('dateLabel').textContent = fechaLarga.charAt(0).toUpperCase() + fechaLarga.slice(1);
  document.getElementById('headerDate').textContent = fechaCorta;

  function selectTurno(turno) {
    turnoSeleccionado = turno;
    document.getElementById('btnManana').classList.toggle('active', turno === 'Mañana');
    document.getElementById('btnTarde').classList.toggle('active', turno === 'Tarde');
  }

  function fmt(val) {
    const n = parseFloat(val) || 0;
    return '$' + n.toLocaleString('es-AR', { minimumFractionDigits: 2, maximumFractionDigits: 2 });
  }

  function enviarWhatsApp() {
    if (!turnoSeleccionado) {
      alert('Por favor seleccioná el turno antes de enviar.');
      return;
    }

    const totalCaja     = document.getElementById('totalCaja').value;
    const ventasPostnet = document.getElementById('ventasPostnet').value;
    const ventasEfectivo= document.getElementById('ventasEfectivo').value;
    const cajaRetirar   = document.getElementById('cajaRetirar').value;
    const cambioSig     = document.getElementById('cambioSiguiente').value;

    const mensaje =
`🧾 *CIERRE DE CAJA*
📅 Fecha: ${fechaCorta}
🕐 Turno: ${turnoSeleccionado}

💰 Total en caja: ${fmt(totalCaja)}
💳 Ventas Postnet: ${fmt(ventasPostnet)}
💵 Ventas efectivo: ${fmt(ventasEfectivo)}

🏧 Caja retirada: ${fmt(cajaRetirar)}
🔄 Cambio turno siguiente: ${fmt(cambioSig)}`;

    // Número destino: 11 7827 6588 → código Argentina 54
    const numero = '5491178276588';
    const url = `https://wa.me/${numero}?text=${encodeURIComponent(mensaje)}`;

    // Toast
    const toast = document.getElementById('toast');
    toast.classList.add('show');
    setTimeout(() => toast.classList.remove('show'), 2500);

    setTimeout(() => window.open(url, '_blank'), 300);
  }
</script>
